function areaCuadrado (lado: REAL ): REAL
type
miArray = array [ 0 .. 5 ] of INTEGER ;
var
resultado: REAL ;

begin
resultado:=+0.0;
miArray[1]:=0;
resultado:=lado*lado+miArray[1+2];
areaCuadrado:=resultado;
end

procedure intercambio (v2,v1: INTEGER )
var
aux: INTEGER ;

begin
if (v1!=v2) then
aux:=0;
aux:=v1;
v1:=v2;
v2:=aux;
end
